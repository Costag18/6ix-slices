# 6ix Slices Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a pizza deal comparison web app for Toronto that compares 7 chains, with group-size calculation, two comparison modes, and on-demand scraping.

**Architecture:** Next.js 15 App Router single-page app. Turso SQLite for cached pricing data. Per-chain scraper modules called from API routes on demand. Client orchestrates freshness checks and parallel refresh calls. Tier 2 chains scraped via GitHub Actions with Puppeteer.

**Tech Stack:** Next.js 15, TypeScript, Tailwind CSS, Turso (libsql), Cheerio, Framer Motion

import { createClient, type Client } from "@libsql/client";
import { initializeSchema, seedChains } from "./schema";
import type { Chain, Pizza, Deal, ScrapeLog } from "./types";

let _client: Client | null = null;

export function getDbClient(): Client {
  if (_client) return _client;
  _client = createClient({
    url: process.env.TURSO_DATABASE_URL!,
    authToken: process.env.TURSO_AUTH_TOKEN!,
  });
  return _client;
}

export async function initializeDb(client: Client): Promise<void> {
  await initializeSchema(client);
  await seedChains(client);
}

export async function getChains(client: Client): Promise<Chain[]> {
  const result = await client.execute("SELECT * FROM chains");
  return result.rows.map((row) => ({
    id: row.id as string,
    name: row.name as string,
    logoUrl: (row.logo_url as string) || "",
    websiteUrl: (row.website_url as string) || "",
  }));
}

export async function upsertPizzas(
  client: Client,
  pizzas: Pizza[]
): Promise<void> {
  for (const pizza of pizzas) {
    await client.execute({
      sql: `INSERT OR REPLACE INTO pizzas
            (id, chain_id, name, size, slices, price, toppings_included, square_inches, category, last_updated)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      args: [
        pizza.id,
        pizza.chainId,
        pizza.name,
        pizza.size,
        pizza.slices,
        pizza.price,
        pizza.toppingsIncluded,
        pizza.squareInches,
        pizza.category,
        pizza.lastUpdated,
      ],
    });
  }
}

export async function upsertDeals(
  client: Client,
  deals: Deal[]
): Promise<void> {
  for (const deal of deals) {
    await client.execute({
      sql: `INSERT OR REPLACE INTO deals
            (id, chain_id, name, description, price, items_included, min_people, max_people, promo_code, valid_days, valid_hours, last_updated)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      args: [
        deal.id,
        deal.chainId,
        deal.name,
        deal.description,
        deal.price,
        JSON.stringify(deal.itemsIncluded),
        deal.minPeople,
        deal.maxPeople,
        deal.promoCode,
        deal.validDays ? JSON.stringify(deal.validDays) : null,
        deal.validHours,
        deal.lastUpdated,
      ],
    });
  }
}

export async function getPizzasByChains(
  client: Client,
  chainIds: string[]
): Promise<Pizza[]> {
  const placeholders = chainIds.map(() => "?").join(",");
  const result = await client.execute({
    sql: `SELECT * FROM pizzas WHERE chain_id IN (${placeholders})`,
    args: chainIds,
  });
  return result.rows.map((row) => ({
    id: row.id as string,
    chainId: row.chain_id as string,
    name: row.name as string,
    size: row.size as Pizza["size"],
    slices: row.slices as number,
    price: row.price as number,
    toppingsIncluded: (row.toppings_included as number) || 0,
    squareInches: (row.square_inches as number) || null,
    category: (row.category as Pizza["category"]) || "specialty",
    lastUpdated: row.last_updated as string,
  }));
}

export async function getDealsByChains(
  client: Client,
  chainIds: string[]
): Promise<Deal[]> {
  const placeholders = chainIds.map(() => "?").join(",");
  const result = await client.execute({
    sql: `SELECT * FROM deals WHERE chain_id IN (${placeholders})`,
    args: chainIds,
  });
  return result.rows.map((row) => ({
    id: row.id as string,
    chainId: row.chain_id as string,
    name: row.name as string,
    description: (row.description as string) || null,
    price: (row.price as number) || null,
    itemsIncluded: JSON.parse((row.items_included as string) || "[]"),

    maxPeople: (row.max_people as number) || null,
    promoCode: (row.promo_code as string) || null,
    validDays: row.valid_days
      ? JSON.parse(row.valid_days as string)
      : null,
    validHours: (row.valid_hours as string) || null,
    lastUpdated: row.last_updated as string,
  }));
}

export async function getLastUpdated(
  client: Client
): Promise<Record<string, string | null>> {
  const chains = await client.execute("SELECT id FROM chains");
  const result: Record<string, string | null> = {};

  for (const chain of chains.rows) {
    const log = await client.execute({
      sql: `SELECT scraped_at FROM scrape_logs
            WHERE chain_id = ? AND status = 'success'
            ORDER BY scraped_at DESC LIMIT 1`,
      args: [chain.id as string],
    });
    result[chain.id as string] =
      log.rows.length > 0 ? (log.rows[0].scraped_at as string) : null;
  }

  return result;
}

export async function logScrape(
  client: Client,
  log: ScrapeLog
): Promise<void> {
  await client.execute({
    sql: `INSERT INTO scrape_logs (chain_id, status, error_message, scraped_at)
          VALUES (?, ?, ?, ?)`,
    args: [log.chainId, log.status, log.errorMessage, log.scrapedAt],
  });
}

import type {
  Pizza,
  Deal,
  Chain,
  AppetiteLevel,
  ChainRecommendation,
} from "./types";
import { SLICES_PER_PERSON } from "./types";

export function calculatePizzasNeeded(
  groupSize: number,
  appetite: AppetiteLevel,
  pizzas: Pizza[]
): { pizza: Pizza; quantity: number }[] {
  const totalSlices = groupSize * SLICES_PER_PERSON[appetite];

  // Pick the pizza with the lowest cost per slice
  const sorted = [...pizzas].sort(
    (a, b) => a.price / a.slices - b.price / b.slices
  );
  const bestPizza = sorted[0];
  if (!bestPizza) return [];

  const quantity = Math.ceil(totalSlices / bestPizza.slices);
  return [{ pizza: bestPizza, quantity }];
}

export function calculatePricePerPerson(
  pizzasNeeded: { pizza: Pizza; quantity: number }[],
  dealsApplied: Deal[],
  groupSize: number
): { totalCost: number; costPerPerson: number } {
  // Calculate base cost from pizzas
  let totalCost = pizzasNeeded.reduce(
    (sum, item) => sum + item.pizza.price * item.quantity,
    0
  );

  // If a deal covers the order for less, use the deal price
  for (const deal of dealsApplied) {
    if (deal.price !== null && deal.price < totalCost) {
      totalCost = deal.price;
    }
  }

  return {
    totalCost: Math.round(totalCost * 100) / 100,
    costPerPerson: Math.round((totalCost / groupSize) * 1000) / 1000,
  };
}

export function calculateValueScore(
  pizza: Pizza,
  activeDeal: Deal | null
): number {
  // Price per square inch (40% weight) — lower is better
  // Range: ~$0.05/sqin (great) to ~$0.20/sqin (bad)
  const squareInches = pizza.squareInches || estimateSquareInches(pizza.size);
  const pricePerSqIn = pizza.price / squareInches;
  const priceScore = Math.max(0, Math.min(100, (0.20 - pricePerSqIn) * 667));

  // Toppings included (20% weight) — more is better
  // Range: 0 (plain) to 5+ (loaded)
  const toppingScore = Math.min(100, pizza.toppingsIncluded * 20);

  // Active promo discount (25% weight)
  let promoScore = 0;
  if (activeDeal?.price !== null && activeDeal?.price !== undefined) {
    const discount = 1 - activeDeal.price / pizza.price;
    promoScore = Math.max(0, Math.min(100, discount * 200));
  }

  // Bundled extras (15% weight) — deals with more items score higher
  let bundleScore = 0;
  if (activeDeal) {
    bundleScore = Math.min(100, activeDeal.itemsIncluded.length * 25);
  }

  const score =
    priceScore * 0.4 +
    toppingScore * 0.2 +
    promoScore * 0.25 +
    bundleScore * 0.15;

  return Math.round(Math.max(0, Math.min(100, score)));
}

function estimateSquareInches(size: string): number {
  const diameters: Record<string, number> = {
    small: 10,
    medium: 12,
    large: 14,
    xlarge: 16,
    party: 18,
  };
  const d = diameters[size] || 14;
  return Math.PI * (d / 2) ** 2;
}

export function rankByPricePerPerson(
  chains: Chain[],
  pizzas: Pizza[],
  deals: Deal[],
  groupSize: number,
  appetite: AppetiteLevel
): ChainRecommendation[] {
  return chains
    .map((chain) => {
      const chainPizzas = pizzas.filter((p) => p.chainId === chain.id);
      const chainDeals = deals.filter((d) => d.chainId === chain.id);
      const pizzasNeeded = calculatePizzasNeeded(
        groupSize,
        appetite,
        chainPizzas
      );
      const { totalCost, costPerPerson } = calculatePricePerPerson(
        pizzasNeeded,
        chainDeals,
        groupSize
      );

      // Calculate average value score for the chain's best pizza
      const bestPizza = pizzasNeeded[0]?.pizza;
      const valueScore = bestPizza
        ? calculateValueScore(bestPizza, chainDeals[0] || null)
        : 0;

      return {
        chain,
        totalCost,
        costPerPerson,
        pizzasNeeded,
        dealsApplied: chainDeals,
        valueScore,
      };
    })
    .filter((r) => r.pizzasNeeded.length > 0)
    .sort((a, b) => a.costPerPerson - b.costPerPerson);
}

export function rankByValueScore(
  chains: Chain[],
  pizzas: Pizza[],
  deals: Deal[],
  groupSize: number,
  appetite: AppetiteLevel
): ChainRecommendation[] {
  const ranked = rankByPricePerPerson(
    chains,
    pizzas,
    deals,
    groupSize,
    appetite
  );
  return ranked.sort((a, b) => b.valueScore - a.valueScore);
}

const SIZE_MAP: Record<string, PizzaSize> = {
  "10": "small",
  "12": "medium",
  "14": "large",
  "16": "xlarge",
  "18": "party",
};

  small: 6,
  medium: 8,
  large: 8,
  xlarge: 10,
  party: 12,
};

const DIAMETER_MAP: Record<PizzaSize, number> = {
  small: 10,
  medium: 12,
  large: 14,
  xlarge: 16,
  party: 18,
};

export function normalizeSizeCode(code: string): PizzaSize {
  return SIZE_MAP[code] || "medium";
}

export function normalizeSizeName(name: string): PizzaSize {
  const lower = name.toLowerCase();
  if (lower.includes("party") || lower.includes("jumbo")) return "party";
  if (lower.includes("extra large") || lower.includes("xlarge") || lower.includes("x-large")) return "xlarge";
  if (lower.includes("large")) return "large";
  if (lower.includes("medium")) return "medium";
  if (lower.includes("small")) return "small";
  return "medium";
}

export function getSlicesForSize(size: PizzaSize): number {
  return SLICES_MAP[size];
}

export function getSquareInches(size: PizzaSize): number {
  const d = DIAMETER_MAP[size];
  return Math.round(Math.PI * (d / 2) ** 2 * 100) / 100;
}

import type { Pizza, Deal } from "@/lib/types";
import {
  normalizeSizeCode,
  getSlicesForSize,
  getSquareInches,
} from "./normalize";

const DOMINOS_STORE_ID = "10391"; // Toronto store
const MENU_URL = `https://order.dominos.ca/power/store/${DOMINOS_STORE_ID}/menu?lang=en&structured=true`;

interface DominosVariant {
  Code: string;
  Name: string;
  Price: string;
  SizeCode: string;
  ProductCode: string;
}

interface DominosCoupon {
  Code: string;
  Name: string;
  Price: string;
  Description?: string;
}

interface DominosApiResponse {
  Variants: Record<string, DominosVariant>;
  Coupons: Record<string, DominosCoupon>;
  Products: Record<string, { Code: string; Name: string; ProductType: string }>;
}

export function parseDominosMenu(data: DominosApiResponse): {
  pizzas: Pizza[];
  deals: Deal[];
} {
  const now = new Date().toISOString();
  const pizzas: Pizza[] = [];
  const deals: Deal[] = [];

  // Parse variants into pizzas
  for (const [code, variant] of Object.entries(data.Variants)) {
    const product = data.Products[variant.ProductCode];
    if (!product || product.ProductType !== "Pizza") continue;

    const size = normalizeSizeCode(variant.SizeCode);
    const price = parseFloat(variant.Price);
    if (isNaN(price) || price <= 0) continue;

    pizzas.push({
      id: `dominos-${code.toLowerCase()}`,
      chainId: "dominos",
      name: variant.Name,
      size,
      slices: getSlicesForSize(size),
      price,
      toppingsIncluded: 0,
      squareInches: getSquareInches(size),
      category: "custom",
      lastUpdated: now,
    });
  }

  // Parse coupons into deals
  for (const [code, coupon] of Object.entries(data.Coupons)) {
    deals.push({
      id: `dominos-coupon-${code}`,
      chainId: "dominos",
      name: coupon.Name,
      description: coupon.Description || null,
      price: parseFloat(coupon.Price) || null,
      itemsIncluded: [coupon.Name],
      minPeople: null,
      maxPeople: null,
      promoCode: code,
      validDays: null,
      validHours: null,
      lastUpdated: now,
    });
  }

  return { pizzas, deals };
}

export async function scrapeDominos(): Promise<{
  pizzas: Pizza[];
  deals: Deal[];
}> {
  const response = await fetch(MENU_URL, {
    headers: {
      "User-Agent": "Mozilla/5.0 (compatible; 6ixSlices/1.0)",
      Accept: "application/json",
    },
  });

  if (!response.ok) {
    throw new Error(`Dominos API returned ${response.status}`);
  }

  const data: DominosApiResponse = await response.json();
  return parseDominosMenu(data);
}

import * as cheerio from "cheerio";
import type { Pizza, Deal } from "@/lib/types";
import { normalizeSizeName, getSlicesForSize, getSquareInches } from "./normalize";

const PIZZA_NOVA_URLS = [
  "https://www.pizzanova.com/products/4505", // Signature
  "https://www.pizzanova.com/products/4513", // Create Your Own
  "https://www.pizzanova.com/products/4494", // Other Favourites
];

interface PizzaNovaProduct {
  productID: number;
  sku: string;
  name: string;
  description: string;
  isVegetarian: boolean;
  recipeType: string;
  sizes: { size: string; price: number; slices: number }[];
}

interface PizzaNovaNextData {
  props: {
    pageProps: {
      products: PizzaNovaProduct[];
    };
  };
}

export function parsePizzaNovaData(data: PizzaNovaNextData): {
  pizzas: Pizza[];
  deals: Deal[];
} {
  const now = new Date().toISOString();
  const pizzas: Pizza[] = [];

  const products = data.props?.pageProps?.products || [];

  for (const product of products) {
    if (product.recipeType !== "PIZZA") continue;

    for (const sizeInfo of product.sizes) {
      const size = normalizeSizeName(sizeInfo.size);
      pizzas.push({
        id: `pizzanova-${product.sku}-${size}`,
        chainId: "pizzanova",
        name: product.name,
        size,
        slices: sizeInfo.slices || getSlicesForSize(size),
        price: sizeInfo.price,
        toppingsIncluded: product.name.toLowerCase().includes("cheese") ? 1 : 3,
        squareInches: getSquareInches(size),
        category: categorizeByName(product.name),
        lastUpdated: now,
      });
    }
  }

  return { pizzas, deals: [] };
}

function categorizeByName(name: string): Pizza["category"] {
  const lower = name.toLowerCase();
  if (lower.includes("cheese") && !lower.includes("pepperoni")) return "cheese";
  if (lower.includes("pepperoni")) return "pepperoni";
  if (lower.includes("custom") || lower.includes("create")) return "custom";
  return "specialty";
}

export async function scrapePizzaNova(): Promise<{
  pizzas: Pizza[];
  deals: Deal[];
}> {
  const allPizzas: Pizza[] = [];

  for (const url of PIZZA_NOVA_URLS) {
    const response = await fetch(url, {
      headers: {
        "User-Agent": "Mozilla/5.0 (compatible; 6ixSlices/1.0)",
      },
    });

    if (!response.ok) continue;

    const html = await response.text();
    const $ = cheerio.load(html);
    const nextDataScript = $("#__NEXT_DATA__").html();

    if (!nextDataScript) continue;

    const data: PizzaNovaNextData = JSON.parse(nextDataScript);
    const { pizzas } = parsePizzaNovaData(data);
    allPizzas.push(...pizzas);
  }

  // Deduplicate by id
  const uniquePizzas = Array.from(
    new Map(allPizzas.map((p) => [p.id, p])).values()
  );

  return { pizzas: uniquePizzas, deals: [] };
}

```typescript
import * as cheerio from "cheerio";
import type { Pizza, Deal } from "@/lib/types";
import { normalizeSizeName, getSlicesForSize, getSquareInches } from "./normalize";

const PIZZAIOLO_URLS = [
  "https://www.pizzaiolo.ca/orders/gourmet-meat-pizzas/categories",
  "https://www.pizzaiolo.ca/orders/gourmet-vegetarian-pizzas/categories",
  "https://www.pizzaiolo.ca/orders/gourmet-vegan-pizzas/categories",
];

export function parsePizzaioloHtml(html: string): {
  pizzas: Pizza[];
  deals: Deal[];
} {
  const $ = cheerio.load(html);
  const now = new Date().toISOString();
  const pizzas: Pizza[] = [];

  $(".product-card").each((_, card) => {
    const name = $(card).find(".product-name").text().trim();
    if (!name) return;

    $(card)
      .find(".price")
      .each((_, priceEl) => {
        const sizeText = $(priceEl).attr("data-size") || $(priceEl).text();
        const priceText = $(priceEl).text().replace("$", "").trim();
        const price = parseFloat(priceText);
        if (isNaN(price)) return;

        const size = normalizeSizeName(sizeText);
        const slug = name.toLowerCase().replace(/[^a-z0-9]+/g, "-");

        pizzas.push({
          id: `pizzaiolo-${slug}-${size}`,
          chainId: "pizzaiolo",
          name,
          size,
          slices: getSlicesForSize(size),
          price,
          toppingsIncluded: 3,
          squareInches: getSquareInches(size),
          category: "specialty",
          lastUpdated: now,
        });
      });
  });

  return { pizzas, deals: [] };
}

export async function scrapePizzaiolo(): Promise<{
  pizzas: Pizza[];
  deals: Deal[];
}> {
  const allPizzas: Pizza[] = [];

  for (const url of PIZZAIOLO_URLS) {
    const response = await fetch(url, {
      headers: {
        "User-Agent": "Mozilla/5.0 (compatible; 6ixSlices/1.0)",
      },
    });

    if (!response.ok) continue;

    const html = await response.text();
    const { pizzas } = parsePizzaioloHtml(html);
    allPizzas.push(...pizzas);
  }

  return { pizzas: allPizzas, deals: [] };
}

The parsing logic (extracting name + price per size) stays the same — only the CSS selectors may need updating.

export async function POST() {
  const githubPat = process.env.GITHUB_PAT;

  if (!githubPat || !githubRepo) {
    return NextResponse.json(
      { error: "GitHub credentials not configured" },
      { status: 500 }
    );
  }

  try {
    const response = await fetch(
      `https://api.github.com/repos/${githubRepo}/actions/workflows/scrape-tier2.yml/dispatches`,
      {
        method: "POST",
        headers: {

          Accept: "application/vnd.github.v3+json",
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ ref: "main" }),
      }
    );

    if (!response.ok) {
      const text = await response.text();
      throw new Error(`GitHub API returned ${response.status}: ${text}`);
    }

    return NextResponse.json({ dispatched: true });
  } catch (error) {
    const message = error instanceof Error ? error.message : "Unknown error";
    return NextResponse.json({ error: message }, { status: 500 });
  }
}

  recommendations: ChainRecommendation[];
  mode: ComparisonMode;
  isLoading: boolean;
  loadingChains: string[];
}

export function ResultsGrid({
  recommendations,
  mode,
  isLoading,
  loadingChains,
}: ResultsGridProps) {
  const [expandedId, setExpandedId] = useState<string | null>(null);

  if (isLoading && recommendations.length === 0) {
    return (
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {Array.from({ length: 3 }).map((_, i) => (
          <SkeletonCard key={i} />
        ))}
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
      <AnimatePresence mode="popLayout">
        {recommendations.map((rec, index) => (
          <motion.div
            key={rec.chain.id}
            layout
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -20 }}
            transition={{ duration: 0.3, delay: index * 0.05 }}
          >
            <DealCard
              recommendation={rec}
              mode={mode}
              rank={index + 1}
              isExpanded={expandedId === rec.chain.id}
              onToggleExpand={() =>
                setExpandedId(
                  expandedId === rec.chain.id ? null : rec.chain.id
                )
              }
            />
          </motion.div>
        ))}
      </AnimatePresence>
      {loadingChains.length > 0 &&
        loadingChains.map((chainId) => <SkeletonCard key={chainId} />)}
    </div>
  );
}

    if (!iso) return "Never";
    const date = new Date(iso);
    return date.toLocaleString("en-CA", {
      month: "short",
      day: "numeric",
      hour: "numeric",
      minute: "2-digit",
    });
  };

  return (
    <footer className="mt-12 py-6 px-4 border-t border-gray-200 text-center text-sm text-[var(--color-text-muted)]">
      <div className="flex flex-wrap justify-center gap-4 mb-3">
        {Object.entries(lastUpdated).map(([chainId, timestamp]) => (
          <span key={chainId}>
            {chainId}: {formatDate(timestamp)}
          </span>
        ))}
      </div>
      <p>Prices may vary by location. Based on Toronto pricing.</p>
    </footer>
  );
}

- [ ] **Step 1: Implement the main page with data fetching and state management**

Replace `src/app/page.tsx` with:

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import { HeroSection } from "@/components/HeroSection";
import { ChainFilter } from "@/components/ChainFilter";
import { ResultsGrid } from "@/components/ResultsGrid";
import { Footer } from "@/components/Footer";
import {
  rankByPricePerPerson,
  rankByValueScore,
} from "@/lib/scoring";
import type {
  AppetiteLevel,
  ComparisonMode,
  Chain,
  Pizza,
  Deal,
  ChainRecommendation,
  FreshnessResponse,
} from "@/lib/types";
import { STALENESS_THRESHOLDS, TIER1_CHAINS, TIER2_CHAINS } from "@/lib/types";

export default function Home() {
  const [groupSize, setGroupSize] = useState(10);
  const [appetite, setAppetite] = useState<AppetiteLevel>("medium");
  const [mode, setMode] = useState<ComparisonMode>("price-per-person");

  const [chains, setChains] = useState<Chain[]>([]);
  const [pizzas, setPizzas] = useState<Pizza[]>([]);
  const [deals, setDeals] = useState<Deal[]>([]);
  const [selectedChains, setSelectedChains] = useState<string[]>([]);
  const [lastUpdated, setLastUpdated] = useState<Record<string, string | null>>(
    {}
  );
  const [loadingChains, setLoadingChains] = useState<string[]>([]);
  const [isInitialLoad, setIsInitialLoad] = useState(true);

  const refreshChain = useCallback(
    async (chainId: string) => {
      setLoadingChains((prev) => [...prev, chainId]);
      try {
        const res = await fetch(`/api/refresh/${chainId}`);
        if (res.ok) {
          // Re-fetch deals to get updated data
          const dealsRes = await fetch(`/api/deals?chains=${chainId}`);
          if (dealsRes.ok) {
            const data = await dealsRes.json();
            setPizzas((prev) => [
              ...prev.filter((p) => p.chainId !== chainId),
              ...data.pizzas,
            ]);
            setDeals((prev) => [
              ...prev.filter((d) => d.chainId !== chainId),
              ...data.deals,
            ]);
            setLastUpdated((prev) => ({
              ...prev,
              [chainId]: new Date().toISOString(),
            }));
          }
        }
      } catch (error) {
        console.error(`Failed to refresh ${chainId}:`, error);
      } finally {
        setLoadingChains((prev) => prev.filter((id) => id !== chainId));
      }
    },
    []
  );

  useEffect(() => {
    async function loadData() {
      try {
        // 1. Check freshness
        const freshnessRes = await fetch("/api/freshness");
        const freshness: FreshnessResponse = await freshnessRes.json();
        setLastUpdated(freshness.chains);

        // 2. Load cached deals
        const dealsRes = await fetch("/api/deals");
        const dealsData = await dealsRes.json();
        setChains(dealsData.chains);
        setPizzas(dealsData.pizzas);
        setDeals(dealsData.deals);
        setSelectedChains(dealsData.chains.map((c: Chain) => c.id));
        setIsInitialLoad(false);

        // 3. Refresh stale Tier 1 chains in parallel
        const now = Date.now();
        const staleTier1 = TIER1_CHAINS.filter((chainId) => {
          const updated = freshness.chains[chainId];
          if (!updated) return true;
          return now - new Date(updated).getTime() > STALENESS_THRESHOLDS.tier1;
        });

        const staleTier2 = TIER2_CHAINS.some((chainId) => {
          const updated = freshness.chains[chainId];
          if (!updated) return true;
          return now - new Date(updated).getTime() > STALENESS_THRESHOLDS.tier2;
        });

        // Refresh stale Tier 1 chains in parallel
        await Promise.all(staleTier1.map((chainId) => refreshChain(chainId)));

        // Dispatch Tier 2 refresh if needed
        if (staleTier2) {
          fetch("/api/refresh/tier2", { method: "POST" }).catch(console.error);
        }
      } catch (error) {
        console.error("Failed to load data:", error);
        setIsInitialLoad(false);
      }
    }

    loadData();
  }, [refreshChain]);

  // Calculate recommendations based on current filters
  const filteredPizzas = pizzas.filter((p) =>
    selectedChains.includes(p.chainId)
  );
  const filteredDeals = deals.filter((d) =>
    selectedChains.includes(d.chainId)
  );
  const filteredChains = chains.filter((c) =>
    selectedChains.includes(c.id)
  );

  const recommendations: ChainRecommendation[] =
    mode === "price-per-person"
      ? rankByPricePerPerson(
          filteredChains,
          filteredPizzas,
          filteredDeals,
          groupSize,
          appetite
        )
      : rankByValueScore(
          filteredChains,
          filteredPizzas,
          filteredDeals,
          groupSize,
          appetite
        );

  const toggleChain = (chainId: string) => {
    setSelectedChains((prev) =>
      prev.includes(chainId)
        ? prev.filter((id) => id !== chainId)
        : [...prev, chainId]
    );
  };

  return (
    <main className="min-h-screen max-w-5xl mx-auto px-4">
      <HeroSection
        groupSize={groupSize}
        appetite={appetite}
        mode={mode}
        onGroupSizeChange={setGroupSize}
        onAppetiteChange={setAppetite}
        onModeChange={setMode}
      />

      <section className="mb-6">
        <ChainFilter
          chains={chains}
          selectedChains={selectedChains}
          onToggleChain={toggleChain}
        />
      </section>

      <ResultsGrid
        recommendations={recommendations}
        mode={mode}
        isLoading={isInitialLoad}
        loadingChains={loadingChains}
      />

      <Footer lastUpdated={lastUpdated} />
    </main>
  );
}
```
